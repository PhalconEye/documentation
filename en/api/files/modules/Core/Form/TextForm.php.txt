class TextForm extends CoreForm
{
    const
        /**
         * Default layout path.
         */
        LAYOUT_TEXT_PATH = 'partials/form/text';

    use EntityForm;

    /**
     * Get layout view path.
     *
     * @return string
     */
    public function getLayoutView()
    {
        return $this->_resolveView(self::LAYOUT_TEXT_PATH);
    }
}
